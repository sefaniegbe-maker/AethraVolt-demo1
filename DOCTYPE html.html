"aiLab.badge": "AI Carbon Lab", "aiLab.title": "Generate Your Strategy", "aiLab.desc": "Input background to let AethraCore build your path.",

"aiLab.placeholder1": "Company Name", "aiLab.placeholder2": "Industry", "aiLab.btn": "Generate Strategy", "aiLab.loading": "Analyzing...",

function changeLang(lang) {

currentLang = lang;

document.getElementById('currentLangLabel').innerText = lang.toUpperCase();

const t = translations[lang] || translations['en'];

document.querySelectorAll('[data-t]').forEach(el => {

const key = el.getAttribute('data-t');

if (t[key]) el.innerText = t[key];

});

 

document.getElementById('inputCompany').placeholder = t['aiLab.placeholder1'] || "Company Name";

document.getElementById('inputIndustry').placeholder = t['aiLab.placeholder2'] || "Industry";

}

 

async function generateAI() {

const company = document.getElementById('inputCompany').value;

const industry = document.getElementById('inputIndustry').value;

if (!company) return;

 

const btn = document.getElementById('aiBtn');

const resultBox = document.getElementById('aiResultContainer');

const resultText = document.getElementById('aiResultText');

const t = translations[currentLang] || translations['en'];

 

btn.disabled = true;

btn.innerHTML = `<i data-lucide="loader-2" class="w-6 h-6 animate-spin"></i> <span>${t['aiLab.loading']}</span>`;

lucide.createIcons();

 

try {

const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

method: 'POST',

headers: { 'Content-Type': 'application/json' },

body: JSON.stringify({

contents: [{ parts: [{ text: `Company: ${company}, Industry: ${industry}` }] }],

systemInstruction: { parts: [{ text: t.sysPrompt }] }

})

});

const data = await response.json();

resultText.innerText = data.candidates?.[0]?.content?.parts?.[0]?.text || "No response";

resultBox.classList.remove('hidden');

} catch (err) {

resultText.innerText = "Error: " + err.message;

resultBox.classList.remove('hidden');

} finally {

btn.disabled = false;

btn.innerHTML = `<i data-lucide="brain-circuit" class="w-6 h-6"></i> <span>${t['aiLab.btn']}</span>`;

lucide.createIcons();

}

}

 

window.addEventListener('scroll', () => {

const nav = document.getElementById('navbar');

const langBtn = document.getElementById('langBtn');

if (window.scrollY > 50) {

nav.classList.add('bg-white/95', 'glass-nav', 'shadow-md', 'py-3', 'text-slate-900');

nav.classList.remove('bg-transparent', 'py-5', 'text-white');

langBtn.classList.add('border-slate-200', 'text-slate-600');

} else {

nav.classList.remove('bg-white/95', 'glass-nav', 'shadow-md', 'py-3', 'text-slate-900');

nav.classList.add('bg-transparent', 'py-5', 'text-white');

langBtn.classList.remove('border-slate-200', 'text-slate-600');

}

});

 

// Initialize Icons

lucide.createIcons();
